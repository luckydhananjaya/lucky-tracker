    <script>

        // 1. SETUP SUPABASE (Replace with your keys!)

        const supabaseUrl = 'sb_publishable_3Yr5BSfr4uHi7X8DfHmNSQ_uQPD-ofH';

        const supabaseKey = 'sb_publishable_3Yr5BSfr4uHi7X8DfHmNSQ_uQPD-ofH';

        const supabase = supabase.createClient(supabaseUrl, supabaseKey);


        // 2. GENERATE QR

        new QRCode(document.getElementById("qrcode"), "Lucky_session_Tracker");


        // 3. ADD SESSIONS LOGIC

        async function addSessions() {

            const email = document.getElementById('clientEmail').value.toLowerCase();

            const amount = parseInt(document.getElementById('amount').value);


            if (!email || !amount) return alert("Please fill both fields");


            // Check if client exists

            const { data: client } = await supabase.from('client_profiles').select('*').eq('email', email).single();


            if (client) {

                const { error } = await supabase.from('client_profiles')

                    .update({ total_sessions: client.total_sessions + amount })

                    .eq('email', email);

                if (!error) alert("Package Added Successfully!");

            } else {

                // If new client, create them

                const { error } = await supabase.from('client_profiles')

                    .insert([{ email, total_sessions: amount, sessions_used: 0 }]);

                if (!error) alert("New Client Created & Package Added!");

            }

        }


        // 4. PWA REGISTER

        if ('serviceWorker' in navigator) { navigator.serviceWorker.register('/sw.js'); }

    </script>
